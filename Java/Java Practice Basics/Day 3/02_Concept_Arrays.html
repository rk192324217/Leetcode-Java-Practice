🔹 Kadane's Finder

public class MainClass {
  public static void main(String[] args) {
    int[] a = {-2, 1, -3, 4, -1, 2, 1, -5, 4};
    int cur=0, mx=a[0];
    for(int v:a){
        cur=Math.max(v, cur+v);
        mx=Math.max(mx,cur);
    }
    System.out.println(mx); // → 6
  }
}
