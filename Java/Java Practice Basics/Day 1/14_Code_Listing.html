08: Max of Three (Ternary)

public class MainClass {
  public static void main(String[] args) {
    int a=10, b=20, c=15;
    int max = (a > b) ? (a > c ? a : c) : (b > c ? b : c);
    System.out.println("Max is: " + max);
  }
}
